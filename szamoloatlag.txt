<script>
    let grades = [];

    function addGrade() {
        let input = document.getElementById("gradeInput");
        let grade = Number(input.value);
        let errorBox = document.getElementById("error");
        errorBox.textContent = ""; // clear old errors

        // Check for decimals
        if (!Number.isInteger(grade)) {
            errorBox.textContent = "Please enter a whole number (no decimals)";
            return;
        }

        // Validation range
        if (grade > 5) {
            errorBox.textContent = "Cannot get a higher grade than 5";
            return;
        }
        if (grade < 1) {
            errorBox.textContent = "Cannot get a grade under 1";
            return;
        }

        // Add grade
        grades.push(grade);

        // Sort grades numerically
        grades.sort((a, b) => a - b);

        // Display grades
        document.getElementById("grades").textContent =
            "Grades (sorted): " + grades.join(", ");

        // Calculate average
        let sum = grades.reduce((a, b) => a + b, 0);
        let avg = (sum / grades.length).toFixed(2);

        document.getElementById("average").textContent =
            "Average: " + avg;

        // Clear input
        input.value = "";
    }
</script>
